function go(id) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('show'));

// 初始化年龄身高学历
function initSelect() {
  let age = document.getElementById('age');
  for(let i=18;i<=50;i++){
    age.innerHTML += `<option value="${i}">${i}岁</option>`;
  }
  let height = document.getElementById('height');
  for(let i=150;i<=200;i++){
    height.innerHTML += `<option value="${i}">${i}cm</option>`;
  }
  let weight = document.getElementById('weight');
  for(let i=40;i<=100;i++){
    weight.innerHTML += `<option value="${i}">${i}kg</option>`;
  }
  let edu = document.getElementById('edu');
  let eds = ["初中","高中","大专","本科","硕士","博士"];
  eds.forEach(e=>{
    edu.innerHTML += `<option value="${e}">${e}</option>`;
  });
}

// 头像预览
function previewAvatar() {
  let file = document.getElementById('avatarUpload').files[0];
  if(!file) return;
  let reader = new FileReader();
  reader.onload = e=>{
    document.getElementById('avatarImg').src = e.target.result;
  }
  reader.readAsDataURL(file);
}

function saveUser() {
  let gender = document.getElementById('gender').value;
  let nick = document.getElementById('nick').value.trim();
  let age = document.getElementById('age').value;
  let height = document.getElementById('height').value;
  let weight = document.getElementById('weight').value;
  let edu = document.getElementById('edu').value;
  let hobby = document.getElementById('hobby').value.trim();
  let wechat = document.getElementById('wechat').value.trim();
  let intro = document.getElementById('intro').value.trim();
  let avatar = document.getElementById('avatarImg').src || "";

  if(!gender || !nick || !age || !height || !wechat || !intro){
    alert("请填写完整信息！");
    return;
  }

  let data = {
    id: Date.now(),
    gender, nick, age, height, weight, edu, hobby, wechat, intro, avatar,
    time: new Date().toLocaleString()
  };

  let list = getUsers();
  list.unshift(data);
  saveUsers(list);

  alert("投稿成功！等待管理员审核");
  document.getElementById('publish').reset();
  document.getElementById('avatarImg').src = "";
  go('home');
}
